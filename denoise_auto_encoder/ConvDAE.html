loss and Optimizer¶

In [93]:
loss = tf.nn.sigmoid_cross_entropy_with_logits(labels=targets_, logits=logits_)
cost = tf.reduce_mean(loss)

optimizer = tf.train.AdamOptimizer(0.001).minimize(cost)

             "Training loss: {:.4f}".format(batch_cost))
